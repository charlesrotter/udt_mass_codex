   215	def covariant_derivative(projector, partial, gamma):
   216	    output = np.asarray(partial).copy()
   217	    for rho, nu, beta, sigma in itertools.product(range(4), repeat=4):
   218	        output[rho,nu,beta] += gamma[nu,rho,sigma]*projector[sigma,beta] - gamma[sigma,rho,beta]*projector[nu,sigma]
   219	    return output

   222	def frobenius(projector, partial, gamma):
   223	    nabla = covariant_derivative(projector, partial, gamma); term = np.zeros((4,4,4))
   224	    for nu, alpha, beta, rho in itertools.product(range(4), repeat=4):
   225	        term[nu,alpha,beta] += projector[rho,alpha]*nabla[rho,nu,beta] - projector[rho,beta]*nabla[rho,nu,alpha]
   226	    value = np.einsum("mn,nab->mab", np.eye(4)-projector, term)
   227	    return float(np.linalg.norm(value)/max(np.linalg.norm(nabla),1e-30))

   117	    for a, b, c, d in itertools.product(range(4), repeat=4):
   118	        weyl[a, b, c, d] = down[a, b, c, d] - 0.5 * (
   119	            g[a, c] * ricci[b, d] - g[a, d] * ricci[b, c]
   120	            - g[b, c] * ricci[a, d] + g[b, d] * ricci[a, c]
   121	        ) + scalar * (g[a, c] * g[b, d] - g[a, d] * g[b, c]) / 6.0
   122	    return {"inverse": inverse, "gamma": gamma, "up": up, "down": down,
   123	            "ricci": ricci, "scalar": scalar, "weyl": weyl}

   217	def primary_projectors(value: np.ndarray) -> tuple[list[np.ndarray], bool]:
   218	    eigenvalues = list(np.linalg.eigvals(value)); clusters: list[list[complex]] = []
   219	    for eigenvalue in sorted(eigenvalues, key=lambda item: (float(item.real), float(item.imag))):
   220	        choices = [index for index, cluster in enumerate(clusters)
   221	                   if abs(eigenvalue - sum(cluster) / len(cluster))
   222	                   <= CLUSTER * max(1.0, abs(eigenvalue), abs(sum(cluster) / len(cluster)))]
   223	        if choices: clusters[choices[0]].append(complex(eigenvalue))
   224	        else: clusters.append([complex(eigenvalue)])
   225	    centers = np.asarray([sum(cluster) / len(cluster) for cluster in clusters], dtype=complex)
   226	    complex_value = np.asarray(value, dtype=complex); identity = np.eye(4, dtype=complex)
   227	    for cluster, center in zip(clusters, centers):
   228	        singular = np.linalg.svd(complex_value - center * identity, compute_uv=False)
   229	        scale = max(1.0, float(singular[0]) if singular.size else 0.0)
   230	        rank = int(np.count_nonzero(singular > TOL * scale))
   231	        uncertain = bool(np.any((singular >= LOW * scale) & (singular <= HIGH * scale)))
   232	        if uncertain or 4 - rank != len(cluster): return [], True
   233	    groups = []; used = set()
   234	    for index, center in enumerate(centers):
   235	        if index in used: continue
   236	        if abs(center.imag) <= CLUSTER * max(1.0, abs(center)): group = [index]
   237	        else:
   238	            partners = [other for other in range(len(centers)) if other != index and other not in used
   239	                        and len(clusters[other]) == len(clusters[index])
   240	                        and abs(centers[other] - center.conjugate())
   241	                        <= CLUSTER * max(1.0, abs(centers[other]), abs(center))]
   242	            if len(partners) != 1: return [], True
   243	            group = [index, partners[0]]
   244	        used.update(group); groups.append(group)
   245	    vandermonde = np.asarray([[center ** power for power in range(len(centers))] for center in centers])
   246	    if np.linalg.cond(vandermonde) > 1.0e12: return [], True
   247	    powers = [np.linalg.matrix_power(complex_value, power) for power in range(len(centers))]
   248	    output = []
   249	    for group in groups:
   250	        target = np.zeros(len(centers), dtype=complex); target[group] = 1.0
   251	        coefficients = np.linalg.solve(vandermonde, target)
   252	        projector = sum((coefficients[power] * powers[power] for power in range(len(powers))),
   253	                        np.zeros((4, 4), dtype=complex))
   254	        if np.max(np.abs(projector.imag)) > TOL: return [], True
   255	        output.append(projector.real)
   256	    return output, False

   265	def central_blocks(values: list[np.ndarray], metric: np.ndarray, commutant, selfcenter):
   266	    ops = normalized(values)
   267	    if len(selfcenter) == 1:
   268	        return [np.eye(4)], "NO_NONTRIVIAL_CENTRAL_BLOCK", False
   269	    candidates = []
   270	    for trial in range(9):
   271	        value = sum((((-1) ** (trial * (index + 1))) * (index + 1) ** (trial + 1)) * item
   272	                    for index, item in enumerate(selfcenter))
   273	        norm = float(np.linalg.norm(value))
   274	        if norm <= ZERO: continue
   275	        projectors, uncertain = primary_projectors(value / norm)
   276	        if uncertain: continue
   277	        valid = True; residual = 0.0
   278	        for projector in projectors:
   279	            rank, rank_uncertain, _ = rank_info(projector)
   280	            current = max(relmax(projector @ projector, projector), relmax(projector.T @ metric, metric @ projector),
   281	                          max((relmax(projector @ op, op @ projector) for op in ops), default=0.0),
   282	                          max((relmax(projector @ op, op @ projector) for op in commutant), default=0.0))
   283	            residual = max(residual, current); valid &= not rank_uncertain and current <= TOL and rank > 0
   284	        residual = max(residual, relmax(sum(projectors, np.zeros((4, 4))), np.eye(4)),
   285	                       max((relmax(projectors[i] @ projectors[j], np.zeros((4, 4)))
   286	                            for i in range(len(projectors)) for j in range(i + 1, len(projectors))), default=0.0))
   287	        if valid and residual <= TOL and sum(rank_info(projector)[0] for projector in projectors) == 4:
   288	            candidates.append(projectors)
   289	    if not candidates: return [], "NUMERIC_UNCERTAIN_CENTRAL_BLOCKS", True
   290	    maximum = max(len(item) for item in candidates); finalists = [item for item in candidates if len(item) == maximum]
   291	    if max((set_distance(finalists[0], item) for item in finalists[1:]), default=0.0) > TOL:
   292	        return finalists[0], "NUMERIC_UNCERTAIN_CENTRAL_BLOCKS", True
   293	    ranks = sorted(rank_info(projector)[0] for projector in finalists[0])
   294	    subset_count = sum(sum(ranks[index] for index in range(len(ranks)) if mask & (1 << index)) == 2
   295	                       for mask in range(1, 2 ** len(ranks) - 1))
   296	    splits = subset_count // 2
   297	    status = "UNIQUE_CENTRAL_2PLUS2" if splits == 1 else "MULTIPLE_CENTRAL_2PLUS2" if splits > 1 \
   298	        else "NO_NONTRIVIAL_CENTRAL_BLOCK" if ranks == [4] else "CENTRAL_BLOCKS_NO_2PLUS2"
   299	    return finalists[0], status, False

   302	def signature(projector: np.ndarray, metric: np.ndarray):
   303	    u, _, _ = np.linalg.svd(projector); rank, uncertain, _ = rank_info(projector); basis = u[:, :rank]
   304	    gram = 0.5 * (basis.T @ metric @ basis + (basis.T @ metric @ basis).T)
   305	    eigenvalues = np.linalg.eigvalsh(gram); scale = max(1.0, float(np.max(np.abs(eigenvalues))))
   306	    negative = int(np.count_nonzero(eigenvalues < -TOL * scale)); positive = int(np.count_nonzero(eigenvalues > TOL * scale))
   307	    zero = rank - negative - positive
   308	    uncertain |= bool(np.any((np.abs(eigenvalues) >= LOW * scale) & (np.abs(eigenvalues) <= HIGH * scale)))
   309	    return rank, f"N{negative}_P{positive}_Z{zero}", negative, positive, zero, basis, uncertain

   324	def orbit(values: list[np.ndarray], seed: np.ndarray, metric: np.ndarray):
   325	    ops = normalized(values)
   326	    if np.linalg.norm(seed) <= ZERO: return 0, "ZERO_SEED", False
   327	    candidates = [seed]; prior = -1; uncertain = False
   328	    while True:
   329	        matrix = np.column_stack([value / np.linalg.norm(value) for value in candidates if np.linalg.norm(value) > ZERO])
   330	        rank, current, _ = rank_info(matrix); uncertain |= current; u, _, _ = np.linalg.svd(matrix, full_matrices=False)
   331	        basis = u[:, :rank]
   332	        if rank == prior or rank == 4: break
   333	        prior = rank; candidates = [basis[:, i] for i in range(rank)] + [op @ basis[:, i] for op in ops for i in range(rank)]
   334	    gram = 0.5 * (basis.T @ metric @ basis + (basis.T @ metric @ basis).T); eigen = np.linalg.eigvalsh(gram)
   335	    scale = max(1.0, float(np.max(np.abs(eigen)))); negative = int(np.count_nonzero(eigen < -TOL * scale)); positive = int(np.count_nonzero(eigen > TOL * scale)); zero = rank - negative - positive
   336	    uncertain |= bool(np.any((np.abs(eigen) >= LOW * scale) & (np.abs(eigen) <= HIGH * scale)))
   337	    return rank, f"N{negative}_P{positive}_Z{zero}", uncertain

   340	def discriminant(operator: np.ndarray, projector: np.ndarray):
   341	    rank, _, _, _, _, basis, _ = signature(projector, np.eye(4))
   342	    if rank != 2: return "NOT_RANK_TWO", False
   343	    restricted = basis.T @ operator @ basis; trace = float(np.trace(restricted)); determinant = float(np.linalg.det(restricted))
   344	    value = trace * trace - 4.0 * determinant; scale = max(1.0, abs(trace * trace), abs(4.0 * determinant)); relative = value / scale
   345	    uncertain = LOW <= abs(relative) <= HIGH
   346	    label = "REAL_DISTINCT" if relative > CLUSTER else "COMPLEX_CONJUGATE" if relative < -CLUSTER else "REPEATED_OR_CLUSTERED"
   347	    return label, uncertain

   364	def classify(values: list[np.ndarray], gradient: np.ndarray, metric: np.ndarray, scalar, keys):
   365	    algebra, ua = algebra_basis(values); commutant, uc = commutant_basis(values); center, uz = center_basis(commutant); selfcenter, us = selfcenter_basis(center, metric)

   377	    if np.linalg.norm(gradient) <= ZERO: incidence = "ZERO_GRADIENT"
   378	    elif unique is None: incidence = "NO_UNIQUE_SPLIT"
   379	    else:

   381	        block_uncertain |= LOW <= minimum <= HIGH
   382	        incidence = "WHOLLY_IN_ONE_PLANE" if minimum <= TOL else "SPLIT_ACROSS_BOTH_PLANES"
   383	    orbit_dimension, orbit_signature, uo = orbit(values, gradient, metric)

   386	    if len(algebra) == 16: reducibility = "FULL_MATRIX_ALGEBRA_IRREDUCIBLE"

   411	    jacobian_jets = [[(j[mu, a], k[mu, a].copy(), ell[mu, a].copy()) for a in range(4)] for mu in range(4)]

# Lay report

We found a promising exact connection inside a conditional geometric branch, but not the whole
emergence story.

The new vocabulary is not just a list of shapes at isolated points. In almost 94,000 of about
95,000 tests, the same geometric type and matchable geometric directions appeared at all 17 sampled
places. That does not prove smooth continuation between the samples. At the midpoints, many
two-dimensional structures had a clear local twisting signal in the chart used for the calculation.
That is the kind of raw material from which linked or knotted global structures might be assembled,
but the assembly has not been derived.

The plain gradient of `phi` behaves differently. It mostly produces ordinary nested surfaces. That
supports the earlier conclusion that `phi`'s gradient by itself is too simple to be the Hopf
soliton's directional field.

The strongest connection appears in the reciprocal angular geometry already found earlier. There,
two metric-derived instruments—the curvature of `phi` and its gradient-direction marker—can identify
the clock direction, the depth direction, and the two angular directions without inserting an
`S2` carrier. Once the two angular directions are additionally made into the correct global circles,
their metric connection produces exactly the same basic Hopf map used to seed the numerical
soliton.

That is significant: after supplying that circle symmetry and global closure, the familiar Hopf
seed can be read exactly as a quotient of reciprocal metric geometry, not merely as a foreign
internal field pasted onto UDT. The audit independently derived the formula from the metric and ran
the repository's actual seed code to check it.

But the metric has not yet chosen the needed global arrangement. Local directions do not tell us
that they are periodic circles, which combination is the Hopf fiber, how the ends close, or why the
complete universe selects this special orderly branch instead of the much more common fully mixed
geometry. Nor have we derived the soliton's energy law.

So the honest result is:

- a fixed-chart, sampled local twisting precursor was observed;
- an exact Hopf-seed compatibility witness exists after explicit reciprocal-toric global premises;
- full carrier and matter emergence remain open.

This gives us a sharper building block, not closure. We still need to learn whether the complete
finite-cell metric selects and sustains the particular global angular assembly that turns its local
motifs into the Hopf quotient and then supplies deformability and dynamics. Until then the proposed
motif-to-Hopf correspondence is a lead.
